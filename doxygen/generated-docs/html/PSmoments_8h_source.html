92 template <int N>
93 void TPSMoments<N>::printFormatted (std::ostream& os, bool normalised) const
94 {
95  using std::setw;
96 
97  int oldp=os.precision(3);
98  ios_base::fmtflags oflg=os.setf(ios::scientific,ios::floatfield);
99 
100  for(int i=0; i<2*N; i++)
101  {
102  os<<setw(12)<<right<<scientific<<this->mean(i);
103  os<<setw(12)<<right<<scientific<<this->std(i);
104  for(int j=1; j<2*N; j++)
105  {
106  if(j<=i)
107  {
108  if(normalised)
109  {
110  os<<setw(8)<<" ";
111  }
112  else
113  {
114  os<<setw(12)<<" ";
115  }
116  }
117  else
118  {
119  os<<right;
120  if(normalised)
121  {
122  os<<setw(8)<<fixed<<this->r_ij(i,j);
123  }
124  else
125  {
126  os<<setw(12)<<scientific<<this->sig(i,j);
127  }
128  }
129  }
130  os<<endl;
131  }
132  os.precision(oldp);
133  os.flags(oflg);
134 }
