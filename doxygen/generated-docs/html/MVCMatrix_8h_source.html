144 template <class T, int N>
145 inline T MVCMatrix<T,N>::r (int n, int m) const
146 {
147  return n==m ? 1 :
148 #ifndef MVCMTRX_REDUCE_MEMORY
149  m2[n][m]/sqrt(variance(n)*variance(m));
150 #else
151  m2[ComputeIndex(n,m)]/sqrt(variance(n)*variance(m));
152 #endif
153 }

213 template <class T, int N>
214 void MVCMatrix<T,N>::printFormatted (ostream& os, bool normalised) const
215 {
216  using std::setprecision;
217  using std::setw;
218  for(int i=0; i<N; i++)
219  {
220  os<<setw(12)<<right<<scientific<<setprecision(3)<<mean(i);
221  os<<setw(12)<<right<<scientific<<setprecision(3)<<std(i);
222  for(int j=1; j<N; j++)
223  {
224  if(j<=i)
225  {
226  if(normalised)
227  {
228  os<<setw(8)<<" ";
229  }
230  else
231  {
232  os<<setw(12)<<" ";
233  }
234  }
235  else
236  {
237  os<<right<<setprecision(3);
238  if(normalised)
239  {
240  os<<setw(8)<<fixed<<r(i,j);
241  }
242  else
243  {
244  os<<setw(12)<<scientific<<operator()(i,j);
245  }
246  }
247  }
248  os<<endl;
249  }
250 }
